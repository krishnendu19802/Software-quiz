create database mainQuiz;

use mainQuiz;

create table users (
userId int AUTO_INCREMENT primary key,
name varchar(255),
email varchar(255),
password varchar(255),
admin int default 0
);


update users set admin=1 where userId=1; // to make first account as admin


create table topics(
topicId int auto_increment primary key,
userId int,
topicName varchar(255)
);



create table questions(
topicId int,
userId int,
questionId int auto_increment primary key,
statement varchar(255),
option1 varchar(255),
option2 varchar(255),
option3 varchar(255),
option4 varchar(255),
ansIndex int default 0
);


CREATE TABLE questionHistory (
    userId INT,
    topicId INT,
    questionId INT,
    status INT DEFAULT 0,
    timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


create table quizHistory(
quizId int auto_increment primary key,
userId int,
timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
score int
);

create table leaderBoard(
userId int ,
score int
);
